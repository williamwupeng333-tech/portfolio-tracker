// 1. 改进后的 XIRR 算法 (使用牛顿迭代法，更精准) function calculateXIRR(cfs) { if (cfs.length < 2) return 0; let rate = 0.1; for (let i = 0; i < 50; i++) { let npv = 0; let dnpv = 0; for (let cf of cfs) { let t = (cf.date - cfs[0].date) / (365 * 24 * 3600 * 1000); let factor = Math.pow(1 + rate, t); npv += cf.amount / factor; dnpv -= (t * cf.amount) / (factor * (1 + rate)); } let newRate = rate - npv / dnpv; if (Math.abs(newRate - rate) < 0.0001) return newRate; rate = newRate; } return rate; } // 2. 从 Google Sheets 抓取实时价格 async function fetchPrices() { const url = "https://docs.google.com/spreadsheets/d/e/2PACX-1vSBHrRHu4C_g6B_aYFSpgPs7uAYGgE4IRlsTa6auBceGHDxEaFO_9yScwF0yoT4cX117cjXI2QeFOIU/pub?gid=0&single=true&output=csv"; // 确保已发布为CSV try { const res = await fetch(url); const text = await res.text(); const lines = text.split(/\r?\n/).slice(1); let prices = {}; lines.forEach(line => { const parts = line.split(","); if (parts.length >= 2) { const ticker = parts[0].replace(/"/g, '').trim().toUpperCase(); const price = parseFloat(parts[1].replace(/"/g, '').trim()); if (!isNaN(price)) prices[ticker] = price; } }); return prices; } catch (e) { console.error("价格获取失败:", e); return {}; } } // 3. 核心计算与显示 async function updateMetrics() { if (trades.length === 0) { document.getElementById("metrics").innerHTML = "暂无交易记录"; return; } const prices = await fetchPrices(); // 生成历史现金流 (买入为负，卖出为正) let cfs = trades.map(t => ({ date: new Date(t.date), amount: t.type === "BUY" ? -(t.shares * t.price) : (t.shares * t.price) })).sort((a, b) => a.date - b.date); // 计算当前持仓数量 let inventory = {}; trades.forEach(t => { inventory[t.ticker] = (inventory[t.ticker] || 0) + (t.type === "BUY" ? t.shares : -t.shares); }); // 计算当前未卖出部分的“实时市值” let currentPortfolioValue = 0; let priceDetailsHTML = "
    "; for (let ticker in inventory) { const shares = inventory[ticker]; if (shares > 0) { const livePrice = prices[ticker] || 0; const marketValue = shares * livePrice; currentPortfolioValue += marketValue; priceDetailsHTML += `
  • ${ticker}: ${shares}股 @ 实时价 ${livePrice.toFixed(2)}
    • `; } } priceDetailsHTML += "
"; // 【关键步】将当前市值作为“今天”的现金流入，存入临时数组计算 IRR const finalCfs = [...cfs, { date: new Date(), amount: currentPortfolioValue }]; const irr = calculateXIRR(finalCfs); // 更新界面 document.getElementById("metrics").innerHTML = `
当前持仓实时市值
￥${currentPortfolioValue.toLocaleString(undefined, {minimumFractionDigits: 2})}
年化 IRR (含未实现盈亏)
${(irr * 100).toFixed(2)}%
${priceDetailsHTML} ${irr < 0.17 ? '
⚠️ 低于 17% 预设目标
' : ''}
`; drawChart(finalCfs); }
