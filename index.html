// 优化的 XIRR 算法 (牛顿迭代法) function calculateXIRR(cfs) { if (cfs.length < 2) return 0; let rate = 0.1; // 初始猜测 10% for (let i = 0; i < 50; i++) { let npv = 0; let dnpv = 0; for (let cf of cfs) { // 计算距离第一笔交易的天数 let t = (cf.date - cfs[0].date) / (365 * 24 * 3600 * 1000); let factor = Math.pow(1 + rate, t); npv += cf.amount / factor; dnpv -= (t * cf.amount) / (factor * (1 + rate)); } let newRate = rate - npv / dnpv; if (Math.abs(newRate - rate) < 0.0001) return newRate; rate = newRate; } return rate; } // 优化的数据拉取 async function fetchPrices() { const url = "你的GoogleSheet发布链接"; try { const res = await fetch(url); const text = await res.text(); const lines = text.split(/\r?\n/).slice(1); let prices = {}; lines.forEach(line => { const parts = line.split(","); if (parts.length >= 2) { const ticker = parts[0].replace(/"/g, '').trim().toUpperCase(); const price = parseFloat(parts[1].replace(/"/g, '').trim()); if (!isNaN(price)) prices[ticker] = price; } }); return prices; } catch (e) { console.error("价格获取失败", e); return {}; } } // 核心指标更新 async function updateMetrics() { if (trades.length === 0) return; const prices = await fetchPrices(); // 1. 生成历史现金流 (注意：BUY 是负向现金流) let cfs = trades.map(t => ({ date: new Date(t.date), amount: t.type === "BUY" ? -(t.shares * t.price) : (t.shares * t.price) })).sort((a, b) => a.date - b.date); // 2. 计算当前持仓市值 let inventory = {}; trades.forEach(t => { inventory[t.ticker] = (inventory[t.ticker] || 0) + (t.type === "BUY" ? t.shares : -t.shares); }); let currentPortfolioValue = 0; for (let ticker in inventory) { if (prices[ticker]) { currentPortfolioValue += inventory[ticker] * prices[ticker]; } } // 3. 将“虚拟平仓市值”加入现金流数组末尾 const finalCfs = [...cfs, { date: new Date(), amount: currentPortfolioValue }]; const irr = calculateXIRR(finalCfs); // 4. UI 渲染 (增加对比你 17% 目标的逻辑) const irrColor = irr >= 0.17 ? "text-green-600" : "text-blue-600"; document.getElementById("metrics").innerHTML = `

当前总市值

￥${currentPortfolioValue.toLocaleString()}

年化 IRR

${(irr*100).toFixed(2)}%

${irr < 0.17 ? `

⚠️ 低于你设定的 17% 目标

` : ''}
`; }
